    def forward(
        self,
        hidden: torch.Tensor,
        target_embeddings: torch.Tensor,  # [b, L-1+K, emb_dim]
        freqs_cis: torch.Tensor,
        logit_scale: torch.Tensor,
        mask: torch.Tensor | None = None,
    ) -> tuple[list[torch.Tensor], torch.Tensor]:
        logits_list    = []
        h_prev         = hidden
        total_aux_loss = torch.tensor(0.0, device=hidden.device, dtype=hidden.dtype)
        seq_len        = hidden.size(1)

        # Labels for main head are tokens[1:seq_len]. 
        # Labels for MTP head k are tokens[k+2:seq_len+k+1].
        # Each head predicts at a distance: head k uses H_t and Emb(t+k+1).
        
        for k in range(self.num_heads):
            # Target for head k is Emb(t+k+1). 
            # target_embeddings was padded, it starts with tokens[1:].
            # index (t+k) in target_embeddings is token t+k+1.
            
            # Valid sequence length for this head: 
            # We need to stay within h_prev and target_embeddings.
            # h_prev has L tokens. target_embeddings has L-1+K.
            # head k: uses H_t and Emb(t+k+1). 
            # available range for H_t: 0 to L-1
            # available range for Emb(t+k+1): 0+k to (L-1+K)-1 = L-2+K
            # To predict T(t+k+2), we need L-1 to have the token.
            
            # Correct alignment:
            # target (Emb_t+k+1) needs to be length of current hidden.
            # We slice target_embeddings starting from index k.
            target = target_embeddings[:, k : k + seq_len]
            
            # Now target has length min(seq_len, total_targets - k)
            current_h_prev = h_prev[:, :target.size(1)]
            
            if current_h_prev.size(1) == 0:
                break

            h_head = current_h_prev + target

            if self.use_full_block:
                head   = self.blocks[k]
                h_head = head["norm"](h_head)
                h_head, aux = head["block"](
                    h_head,
                    freqs_cis[: h_head.size(1)],
                    mask=(mask[:, :, : h_head.size(1), : h_head.size(1)]
                          if mask is not None else None),
                )
                h_head = head["out_norm"](h_head)
            else:
                head   = self.ffn_layers[k]
                h_head = head["norm"](h_head)
                h_head = head["ffn"](h_head)
                h_head = head["out_norm"](h_head)
                aux    = torch.tensor(0.0, device=hidden.device, dtype=hidden.dtype)

            logits = self.proj(h_head) * logit_scale
            logits_list.append(logits)

            # Recurse: next head builds on this head's output
            h_prev         = h_head
            total_aux_loss = total_aux_loss + aux

        return logits_list, total_aux_loss
